#+begin_src sh
  if [ $TMUX ]; then
      current_tmux="$(tmux display-message -p '#W/#P')"
      echo "Current tmux-session: $current_tmux"
      target_script="$HOME/lib/tmux-scripts/$current_tmux.sh"
      if [ -f $target_script ]; then
          sh $target_script
      fi
  fi
